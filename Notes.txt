== New class scheme:
There are two signatures for each typeclass: X and XInstance. Example:

signature Monad =
sig
  type 'a t
  val >>= : 'a t * ('a -> 'b t) -> 'b t
  val return : 'a -> 'a t
end

signature MonadInstance =
sig
  include Monad
  val mapM : ...
  ...
end

Usage:
Say List is a monad. Then:

signature List =
sig
  type 'a t
  (* Monad *)
  val >>= : 'a t * ('a -> 'b t) -> 'b t
  val return : 'a -> 'a t
  (* Alternative *)
  ...
end

structure List =
struct (Monad)
  ...
end
